### [69.x的平方根](https://leetcode-cn.com/problems/sqrtx/description/)
```c++
class Solution {
public:
    int mySqrt(int x) {
        if (x < 0) return INT_MIN;
        if (x == 0) return 0;
        double pre = 0, ans = 1;
        while(abs(ans-pre) > 1e-6) {
            pre = ans;
            ans = (ans+x/ans)/2;
        }
        return int(ans);
    }
};
```
